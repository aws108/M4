2. Retorna el títol d'aquells llibres de categoria "web" que tinguin un preu inferior a 40€.
<llibres>
{
	for $llibres in //book
	let $categoria := $llibres/@category
	let $precio := $llibres/price
	where ($categoria ="web" and $precio<=40)
	return $llibres/title/text()
}
</llibres>
